// Developer Dashboard Login Persistence Fix const ACCESS_CODE = "YOUR_SECRET_CODE"; // Replace with your real code // Run on page load window.onload = function () { if (localStorage.getItem("isLoggedIn") === "true") { showDashboard(); } else { showHome(); } }; function authenticate() { const code = document.getElementById("accessCode").value; if (code === ACCESS_CODE) { localStorage.setItem("isLoggedIn", "true"); // Save login state showDashboard(); } else { document.getElementById("errorMessage").classList.add("show"); } } function logout() { localStorage.removeItem("isLoggedIn"); // Clear login state showHome(); } function showDashboard() { document.getElementById("loginScreen").classList.remove("active"); document.getElementById("guestPage").classList.add("hidden"); document.getElementById("dashboard").classList.add("active"); } function showHome() { document.getElementById("loginScreen").classList.remove("active"); document.getElementById("guestPage").classList.remove("hidden"); document.getElementById("dashboard").classList.remove("active"); }
